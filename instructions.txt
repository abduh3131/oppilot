Watch the data (WSL, Terminal #2)

Open a second Ubuntu window and paste:

cd ~/dev/op-jetson-adapters
source .venv/bin/activate
python - << 'PY'
import zmq, json
ctx=zmq.Context.instance()
s=ctx.socket(zmq.SUB); s.connect("tcp://127.0.0.1:5557")
for t in (b"gps.raw", b"camera.front", b"camera.front.bytes"):
    s.setsockopt(zmq.SUBSCRIBE, t)
print("listening… (Ctrl+C to stop)")
while True:
    parts = s.recv_multipart()
    topic = parts[0].decode()
    if topic.endswith(".bytes"):
        print(topic, "bytes=", len(parts[1]))
    else:
        print(topic, json.loads(parts[1]))
PY

Optional: save a camera frame to a PNG (quick visual)

cd ~/dev/op-jetson-adapters
source .venv/bin/activate
python - << 'PY'
import zmq, json, numpy as np
from PIL import Image
ctx=zmq.Context.instance(); s=ctx.socket(zmq.SUB)
s.connect("tcp://127.0.0.1:5557")
for t in (b"camera.front", b"camera.front.bytes"):
    s.setsockopt(zmq.SUBSCRIBE, t)
w=h=None
while True:
    topic, payload = s.recv_multipart()
    if topic==b"camera.front":
        md=json.loads(payload); w,h=md["w"],md["h"]
    elif topic==b"camera.front.bytes" and w and h:
        img=np.frombuffer(payload,dtype=np.uint8).reshape(h,w,3)
        Image.fromarray(img).save("/tmp/mock_frame.png")
        print("Saved /tmp/mock_frame.png"); break
PY


Open it from Windows: \\wsl.localhost\Ubuntu\tmp\mock_frame.png
